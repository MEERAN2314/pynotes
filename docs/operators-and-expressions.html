Operators And Expressions

Mon 30 June 2025 
x, y = 10, 3
print(f"{x} + {y} = {x + y}")  
print(f"{x} - {y} = {x - y}")  
print(f"{x} * {y} = {x * y}")  
print(f"{x} / {y} = {x / y}")  
print(f"{x} // {y} = {x // y}")
print(f"{x} % {y} = {x % y}")  
print(f"{x} ** {y} = {x ** y}") 


print(f"{x} == {y}: {x == y}")  
print(f"{x} != {y}: {x != y}")  
print(f"{x} > {y}: {x > y}")    


print(True and False)  
print(True or False)   
print(not True)
