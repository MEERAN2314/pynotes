print("Counting to 5:")
for i in range(1, 6):
    print(i)


count = 0
print("\nWhile loop example:")
while count < 3:
    print(f"Count is {count}")
    count += 1

print("\nLoop control:")
for num in range(10):
    if num == 2:
        continue  
    if num == 7:
        break     
    print(num)
