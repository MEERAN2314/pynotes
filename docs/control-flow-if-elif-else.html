age = 18
if age >= 18:
    print("You are an adult")
else:
    print("You are a minor")


score = 85
if score >= 90:
    grade = 'A'
elif score >= 80:
    grade = 'B'
elif score >= 70:
    grade = 'C'
else:
    grade = 'D'
print(f"Your grade is {grade}")
